Authentication

All API requests require your API key in the header:

Available Endpoints

GET /api/services Get list of all available services
Response Example:
{
  "status": "success",
  "data": [
    {
      "id": 1,
      "name": "Instagram Followers",
      "category": "instagram",
      "price": 200,
      "min": 100,
      "max": 100000
    }
  ]
}
POST /api/order Place a new order
Request Body:
{
  "service": 1,
  "link": "https://instagram.com/username",
  "quantity": 1000
}
Response Example:
{
  "status": "success",
  "order_id": 12345,
  "message": "Order placed successfully"
}
GET /api/status/{order_id} Check order status
Response Example:
{
  "status": "success",
  "order": {
    "id": 12345,
    "status": "Completed",
    "start_count": 1000,
    "remains": 0
  }
}
GET /api/balance Check account balance
Response Example:
{
  "status": "success",
  "balance": 1500.50,
  "currency": "INR"
}

Code Examples

// Place an order using JavaScript
const response = await fetch('https://api.indiasocialpanel.in/v1/api/order', {
  method: 'POST',
  headers: {
    'Authorization': 'Bearer YOUR_API_KEY',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    service: 1,
    link: 'https://instagram.com/username',
    quantity: 1000
  })
});

const data = await response.json();
console.log(data);
<?php
// Place an order using PHP
$curl = curl_init();

curl_setopt_array($curl, array(
  CURLOPT_URL => 'https://api.indiasocialpanel.in/v1/api/order',
  CURLOPT_RETURNTRANSFER => true,
  CURLOPT_POST => true,
  CURLOPT_POSTFIELDS => json_encode(array(
    'service' => 1,
    'link' => 'https://instagram.com/username',
    'quantity' => 1000
  )),
  CURLOPT_HTTPHEADER => array(
    'Authorization: Bearer YOUR_API_KEY',
    'Content-Type: application/json'
  ),
));

$response = curl_exec($curl);
curl_close($curl);

echo $response;
?>
